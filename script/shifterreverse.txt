:shifterset2
{$USE CLEO}
thread 'shiftrv'
wait 10 
008B: 0@ = $DELOREAN // integer values and handles 
if or
   not Car.Wrecked(0@)
00DC:   player $PLAYER_CHAR driving 0@ 
else_jump @shifterset2      
3FFD: rotate_car_part "rpmneedle" angle 0.0 406.8 0.0 car 0@ //RPM 1500 //Neutral 
3FFB: set_car 0@ component "shiftern" visible 1       //Neutral Gear on
3FFB: set_car 0@ component "shifterr" visible 0      //Reverse Gear off
3FFB: set_car 0@ component "shifter1" visible 0     //1st Gear shifter off
3FFB: set_car 0@ component "shifter2" visible 0    //2nd shifter off
3FFB: set_car 0@ component "shifter3" visible 0   //3rd Gear off
3FFB: set_car 0@ component "shifter4" visible 0  //4th Gear off
3FFB: set_car 0@ component "shifter5" visible 0 //5th Gear off                                          
if 
00E1:   player 0 pressed_button 14
THEN                       
3FFB: set_car 0@ component "shiftern" visible 0       //Neutral Gear off
3FFB: set_car 0@ component "shifterr" visible 1      //Reverse Gear on
3FFB: set_car 0@ component "shifter1" visible 0     //1st Gear shifter off
3FFB: set_car 0@ component "shifter2" visible 0    //2nd shifter off
3FFB: set_car 0@ component "shifter3" visible 0   //3rd Gear off
3FFB: set_car 0@ component "shifter4" visible 0  //4th Gear off
3FFB: set_car 0@ component "shifter5" visible 0 //5th Gear off  
end
jump @shifterset2
