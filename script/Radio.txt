:Radio
thread 'RADIO' 
6@ = 0 // integer values 
1@ = 0 // integer values 

:RadioStart
wait 0 
if 
   Player.Defined($PLAYER_CHAR)
jf @RadioStart 
if 
   Car.Wrecked(0@)
jf @RadioDriving 
if 
00E0:   player $PLAYER_CHAR driving 
jf @RadioStart 
03C1: 0@ = player $PLAYER_CHAR car 
041E: set_radio_station 1@ -1 

:RadioDriving
wait 0 
if 
00E0:   player $PLAYER_CHAR driving 
jf @RadioDriving 
if 
   not Car.Wrecked(0@)
jf @RadioStart 
if 
00DC:   player $PLAYER_CHAR driving 0@ 
jf @RadioReset 
if 
00E1:   key_pressed 0 4 
jf @RadioDriving 
1@ += 1 // integer values 
if 
  1@ > 9 // integer values 
then
  1@ = 0 // integer values
end 

:RadioChange
wait 0 
041E: set_radio_station 1@ -1 
if 
80E1:   not key_pressed 0 4 
jf @RadioChange 
jump @RadioDriving 

:RadioReset
0@ = -1 // integer values 
jump @RadioStart
